弹出层-dialog

需先引入jQuery，然后引入组件 src="./script/M-dialog.js"

弹窗 
$('.dialog1').click(function(){
    $('.M-box').mDialog({
        title:'标题',
        content:'内容',
        clickMask:true,
        button:{
            'yes':function(callback){
                alert('确定前回调');
                callback.close();
            },
            'no':function(callback){
                callback.close();
            }
        },
        beforeShow:function(){
            alert('显示前事件');
        }
    });
});
