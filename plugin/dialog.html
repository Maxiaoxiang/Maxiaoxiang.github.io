弹出层-dialog

需先引入jQuery，然后引入dialog组件

<script src="jquery.js"></script>
<script src="M-dialog.js"></script>

组件样式与功能分离，自定义样式

HTML

<div class="M-box"></div>

Javascript

$('.dialog1').click(function(){
    $('.M-box').mDialog({
        title:'标题',
        content:'内容',
        clickMask:true,
        button:{
            'yes':function(callback){
                alert('确定前回调');
                callback.close();
            },
            'no':function(callback){
                callback.close();
            }
        },
        beforeShow:function(){
            alert('显示前事件');
        }
    });
});

HTML

<div class="M-message"></div>

Javascript

$('.dialog2').click(function(){
    $('.M-message').mDialog({
        type:'message',
        content:'这条消息2秒后关闭',
        beforeShow:function(){
            alert('显示前事件');
        },
        afterHide:function(){
            alert('关闭后事件');
        }
    });
});
